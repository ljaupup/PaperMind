创建 `app/infrastructure/persistence/paper_repository.py`。专题服务只依赖其 `get()` 方法；采集服务复用 `upsert_many()`，因此不会在两个位置重复 SQL：

```python
from collections.abc import Iterable

from sqlalchemy import select

from app.domain.models import Paper
from app.infrastructure.persistence.models import PaperRecord


def to_paper(record: PaperRecord) -> Paper:
    return Paper.model_validate(record, from_attributes=True)


class PostgresPaperRepository:
    def __init__(self, session_factory) -> None:
        self.session_factory = session_factory

    def get(self, paper_id: str) -> Paper | None:
        with self.session_factory() as session:
            record = session.get(PaperRecord, paper_id)
            return None if record is None else to_paper(record)

    def list(self, *, limit: int) -> list[Paper]:
        with self.session_factory() as session:
            records = session.scalars(
                select(PaperRecord).order_by(PaperRecord.published_at.desc(), PaperRecord.paper_id.desc()).limit(limit)
            ).all()
            return [to_paper(record) for record in records]

    def upsert_many(self, papers: Iterable[Paper]) -> tuple[int, int]:
        created = updated = 0
        with self.session_factory() as session:
            for paper in papers:
                record = session.get(PaperRecord, paper.paper_id)
                values = paper.model_dump()
                if record is None:
                    session.add(PaperRecord(**values)); created += 1
                else:
                    for key, value in values.items(): setattr(record, key, value)
                    updated += 1
            session.commit()
        return created, updated
```

创建 `app/infrastructure/arxiv/collector.py`。该文件只负责把第三方 `arxiv.Result` 转为内部 `Paper`；保存、索引和 HTTP 响应均由其他层负责：

```python
import asyncio

import arxiv

from app.domain.models import Paper

_CLIENT = arxiv.Client(page_size=10, delay_seconds=4.0, num_retries=1)


def clean_text(value: str | None) -> str:
    return " ".join(value.split()) if value else ""


def result_to_paper(result: arxiv.Result) -> Paper:
    return Paper(
        paper_id=result.get_short_id(),
        title=clean_text(result.title),
        abstract=clean_text(result.summary),
        authors=[name for author in result.authors if (name := clean_text(author.name))],
        url=result.entry_id,
        pdf_url=result.pdf_url,
        published_at=result.published,
    )


def _collect_sync(query: str, max_results: int) -> list[Paper]:
    search = arxiv.Search(
        query=query,
        max_results=max_results,
        sort_by=arxiv.SortCriterion.SubmittedDate,
        sort_order=arxiv.SortOrder.Descending,
    )
    try:
        return [result_to_paper(result) for result in _CLIENT.results(search)]
    except arxiv.HTTPError as exc:
        if exc.status == 429:
            raise RuntimeError("arXiv 限流；等待至少 60 秒后仅重试一次") from exc
        raise


async def collect_arxiv(query: str, max_results: int) -> list[Paper]:
    return await asyncio.to_thread(_collect_sync, query, max_results)
```

```python
from app.domain.models import Paper, PaperChunk


def chunk_text(text: str, *, chunk_size: int = 700, overlap: int = 120) -> list[str]:
    if chunk_size <= 0 or not 0 <= overlap < chunk_size:
        raise ValueError("chunk_size must be positive and overlap must be smaller")
    normalized = " ".join(text.split())
    if not normalized:
        return []
    result: list[str] = []
    start = 0
    while start < len(normalized):
        end = min(start + chunk_size, len(normalized))
        result.append(normalized[start:end])
        if end == len(normalized):
            break
        start = end - overlap
    return result


def build_abstract_chunks(papers: list[Paper]) -> list[PaperChunk]:
    chunks: list[PaperChunk] = []
    for paper in papers:
        for index, text in enumerate(chunk_text(f"{paper.title}\n{paper.abstract}")):
            chunks.append(PaperChunk(
                chunk_id=f"{paper.paper_id}-{index}", paper_id=paper.paper_id,
                title=paper.title, url=paper.url, pdf_url=paper.pdf_url, text=text,
            ))
    return chunks
```

创建 `app/infrastructure/ai/embeddings.py`。Hash 实现用于没有 API key 的本地测试；SiliconFlow 走 OpenAI 兼容的 `/embeddings` 接口。两个实现都返回相同的 `list[list[float]]`，上层不需要知道模型来源：

```python
import hashlib
import math
import json
import re
from typing import Protocol

import httpx

from app.core.config import Settings


class EmbeddingClient(Protocol):
    async def embed_texts(self, texts: list[str]) -> list[list[float]]: ...


class HashEmbeddingClient:
    def __init__(self, dimensions: int = 64) -> None:
        self.dimensions = dimensions

    async def embed_texts(self, texts: list[str]) -> list[list[float]]:
        vectors: list[list[float]] = []
        for text in texts:
            vector = [0.0] * self.dimensions
            for token in text.lower().split():
                bucket = int.from_bytes(hashlib.sha256(token.encode()).digest()[:4], "big") % self.dimensions
                vector[bucket] += 1.0
            length = math.sqrt(sum(value * value for value in vector))
            vectors.append([value / length for value in vector] if length else vector)
        return vectors


class OpenAICompatibleEmbeddingClient:
    def __init__(self, api_key: str, base_url: str, model: str) -> None:
        self.api_key, self.base_url, self.model = api_key, base_url.rstrip("/"), model

    async def embed_texts(self, texts: list[str]) -> list[list[float]]:
        if not self.api_key:
            raise RuntimeError("embedding provider requires an API key")
        async with httpx.AsyncClient(timeout=60.0) as client:
            response = await client.post(
                f"{self.base_url}/embeddings",
                headers={"Authorization": f"Bearer {self.api_key}"},
                json={"model": self.model, "input": texts},
            )
            response.raise_for_status()
        return [item["embedding"] for item in response.json()["data"]]


def create_embedding_client(settings: Settings) -> EmbeddingClient:
    if settings.embedding_provider == "siliconflow":
        return OpenAICompatibleEmbeddingClient(
            settings.siliconflow_api_key, settings.siliconflow_base_url,
            settings.siliconflow_embedding_model,
        )
    return HashEmbeddingClient()
```

创建 `app/infrastructure/vector/chroma.py`。元数据随每个向量写入，检索后才能返回论文 URL、PDF 与摘要片段，而不是只有没有出处的相似度分数：

```python
from typing import Any

import chromadb

from app.domain.models import PaperChunk


class ChromaVectorStore:
    def __init__(self, host: str, port: int, collection_name: str) -> None:
        self.collection = chromadb.HttpClient(host=host, port=port).get_or_create_collection(
            name=collection_name,
            metadata={"hnsw:space": "cosine"},
        )

    def upsert_chunks(self, chunks: list[PaperChunk], embeddings: list[list[float]]) -> None:
        if len(chunks) != len(embeddings):
            raise ValueError("chunks and embeddings must have equal length")
        if not chunks:
            return
        self.collection.upsert(
            ids=[chunk.chunk_id for chunk in chunks],
            embeddings=embeddings,
            documents=[chunk.text for chunk in chunks],
            metadatas=[{
                "paper_id": chunk.paper_id, "title": chunk.title, "url": chunk.url,
                "pdf_url": chunk.pdf_url or "", "page": chunk.page or 0,
            } for chunk in chunks],
        )

    def query(self, embedding: list[float], top_k: int, paper_ids: list[str] | None = None) -> list[dict[str, Any]]:
        if paper_ids == []:
            return []
        kwargs: dict[str, Any] = {"query_embeddings": [embedding], "n_results": top_k}
        if paper_ids is not None:
            kwargs["where"] = {"paper_id": {"$in": paper_ids}}
        result = self.collection.query(**kwargs)
        documents = result.get("documents", [[]])[0]
        metadatas = result.get("metadatas", [[]])[0]
        distances = result.get("distances", [[]])[0]
        return [
            {"text": text, **metadata, "score": float(distance)}
            for text, metadata, distance in zip(documents, metadatas, distances, strict=True)
        ]
```

```python
    async def ask(self, question: str, *, paper_ids: list[str], top_k: int) -> tuple[str, list[dict], bool]:
        sources = await self.retrieval.search(question, paper_ids=paper_ids, top_k=top_k)
        if not sources:
            return "证据不足，无法回答。", [], True
        return await self.llm.generate(question, [item["text"] for item in sources]), sources, False
```

## 开始前：从原有 `app/` 到当前分层结构

在开始阶段 1 前，先把一次已经完成的后端整理看作“交接”，而不是新的功能任务。原来的 MVP 把路由、编排、模型服务和数据库代码放在同一个 `app/` 目录中；这在功能很少时容易找到文件，但当增加专题、任务、全文和用户数据后，修改一个功能容易牵连多个职责。

**分层**就是按职责给代码划定边界：API 层处理 HTTP 请求，应用层组织一个用例，领域层表达业务规则与接口约定，基础设施层实现 PostgreSQL、arXiv、Chroma 和模型服务。这样更换底层技术时，业务规则不需要随之散落修改。

先看一个与 PaperMind 无关、可以直接运行的最小例子：

```python
class MemoryStore:
    """在内存中保存文本。"""

    def __init__(self) -> None:
        self.items: list[str] = []

    def save(self, text: str) -> None:
        """保存一条文本。"""
        self.items.append(text)


def create_note(store: MemoryStore, text: str) -> None:
    """创建并保存一条笔记。"""
    store.save(text.strip())


store = MemoryStore()
create_note(store, "  first note  ")
assert store.items == ["first note"]
```

这里 `create_note()` 只表达“创建笔记”这个用例；`MemoryStore` 负责“如何保存”。PaperMind 中的 `application` 与 `infrastructure` 也采用相同分工，只是将内存列表替换为可切换的 PostgreSQL、Chroma、arXiv 或模型服务适配器。

### 原文件与新位置的对应关系

下表是学习时最常用的定位表。箭头表示代码职责的去向，不代表每个旧文件都被逐字拆分到唯一的新文件。

| 原 MVP 文件 | 当前位置 | 为什么这样迁移 |
| --- | --- | --- |
| `app/main.py` | `app/main.py`、`app/api/router.py`、`app/api/v1/` | `main.py` 只创建应用并注册 Router；健康检查、论文采集、索引检索和问答分别成为小 Router。 |
| `app/config.py` | `app/core/config.py` | 环境变量与运行配置属于所有层都可读取的核心配置。 |
| `app/schemas.py` | `app/schemas/papers.py`、`knowledge.py`、`conversations.py` | 按 API 资源拆分请求/响应 DTO，避免一个文件同时承载所有接口格式。 |
| `app/collector.py` | `app/infrastructure/arxiv/collector.py` | arXiv SDK 和第三方返回对象是外部技术细节。 |
| `app/storage.py` | `app/infrastructure/persistence/repositories.py` | 本地 JSON Repository 与存储实现选择移入持久化适配器。 |
| `app/db/database.py`、`models.py`、`repositories.py` | `app/infrastructure/persistence/database.py`、`models.py`、`postgres_repository.py` | 数据库连接、ORM 表模型和 PostgreSQL 查询均属于基础设施。原 `app/db/` 不再保留源代码。 |

| `app/vector_store.py` | `app/infrastructure/vector/chroma.py` | Chroma 的 collection、upsert 和 query API 与业务检索规则分离。 |
| `app/indexer.py`、`app/retriever.py`、`app/rag.py`、`app/qa_engine.py` | `app/application/indexing_service.py`、`retrieval_service.py`、`rag_service.py`，并复用 AI 适配器 | 原先分散或重叠的编排职责集中为“建立索引、检索、生成回答”三个用例；`qa_engine.py` 不再单独存在。 |

`app/domain/models.py` 与 `app/domain/ports.py` 是这次新增的关键边界：前者放 `Paper`、`Answer` 等不依赖数据库或 HTTP 的业务数据；后者以 `Protocol` 声明 `PaperRepository`、`EmbeddingClient`、`LLMClient`、`VectorStore` 等“应用需要什么能力”。基础设施层再提供这些能力的真实实现。

### 当前一次请求如何流动

```text
HTTP 请求
→ api/router.py 与 api/v1/            # 匹配路径、校验 HTTP 输入
  → api/dependencies.py                 # 从容器取得所需应用服务
  → application/*_service.py            # 编排“采集、索引、检索、问答”用例
  → domain/models.py 与 ports.py         # 业务数据和抽象能力约定
  → infrastructure/*                    # PostgreSQL、Chroma、arXiv、Embedding、LLM 的真实调用
```

`app/core/container.py` 是**组装根**：它读取 `Settings`，创建 Repository、向量库和模型客户端，再把它们注入应用服务。业务代码不应在 Router 内直接 `new` Chroma 客户端，也不应在领域模型内读取 `.env`；这样单元测试才能用 Fake 实现替代外部服务。

阶段 1 新增 `Topic` 时，沿用这一方向：先在 `domain` 定义专题及其端口，再在 `application` 编排用例，随后在 `infrastructure/persistence` 落地 ORM、迁移和 Repository，最后从 `api` 暴露 RESTful 资源。下文会按此顺序实施。

## 0. 先认识本阶段的三个概念

**持久化**是让数据在程序或浏览器刷新后仍能找回；数据库表保存数据，API 让前端能够读取和修改它。**资源**是可独立识别的业务对象，例如一个专题；RESTful API 用 URL 表示资源、用 HTTP 方法表示动作。**迁移**是数据库结构的版本记录，使新机器能按相同步骤创建同样的表。

先看与项目无关的最小例子：

```python
"""演示内存中的专题资源。"""

topics: list[dict[str, str]] = []


def create_topic(name: str) -> dict[str, str]:
    """创建一个专题字典。

    :return: 包含新专题标识和名称的字典。
    """
    topic = {"id": str(len(topics) + 1), "name": name}
    topics.append(topic)
    return topic
```

调用 `create_topic("RAG")` 会返回 `{"id": "1", "name": "RAG"}`。阶段 1 会把这个仅存于内存的字典，替换为 `Topic` 领域模型、PostgreSQL 表和 `/api/v1/topics` 接口。

### 1.1 用户路径

```text
进入专题列表
  → 创建“生成式虚假信息检测与治理”专题
```

### 1.2 本阶段领域模型

本阶段只新增三个概念：

| 模型 | 必要字段 | 业务规则 |
| --- | --- | --- |
| `Workspace` | `id`、`name`、创建时间 | 单用户固定工作区即可；为未来用户归属预留 `workspace_id`。 |
| `Topic` | `id`、`workspace_id`、`name`、`description`、`keywords`、`categories`、创建/更新时间 | 同一工作区内专题名称唯一；关键词不能为空。 |

现有 `Paper` 仍是全局论文实体。不要在 `papers` 表中加入 `topic_id`，否则一篇论文无法属于多个专题。

### 1.3 RESTful API 目标契约

从本阶段开始新增接口统一使用 `/api/v1`。旧 `/papers/collect`、`/search`、`/ask` 暂保留为 `v0.1` 兼容入口，前端只调用新接口。

| 方法 | 路径 | 含义 | 成功响应 |
| --- | --- | --- | --- |
| `POST` | `/api/v1/topics` | 创建专题 | `201 Created` + `TopicResponse` |
| `GET` | `/api/v1/topics` | 分页查询专题 | `200 OK` |
| `GET` | `/api/v1/topics/{topic_id}` | 查询专题详情 | `200 OK` |
| `PATCH` | `/api/v1/topics/{topic_id}` | 局部更新专题 | `200 OK` |
| `DELETE` | `/api/v1/topics/{topic_id}` | 删除专题及其关联关系 | `204 No Content` |
| `GET` | `/api/v1/topics/{topic_id}/papers` | 查询专题论文 | `200 OK` |
| `POST` | `/api/v1/topics/{topic_id}/papers` | 关联已有论文 | `201 Created` |
| `PATCH` | `/api/v1/topics/{topic_id}/papers/{paper_id}` | 更新专题内收藏状态 | `200 OK` |

下文的第一条纵向切片先实现创建、列表和详情；`PATCH`、删除与专题论文关联必须按同一 DTO、Service、Repository、API 测试顺序补齐后，才可宣称本节目标契约已完成。不要因为前三个接口能运行，就跳过后五个资源操作。

统一约定：

- 本阶段列表先使用确定排序的 `limit`；在为 `(created_at, id)` 写出集成测试前，不暴露看似可用、实际会漏数据的 `cursor`。
- 所有不存在的资源返回 `404`；违反唯一约束返回 `409`；请求字段不合法由 FastAPI 返回 `422`。
- 自定义业务错误使用 `application/problem+json` 风格，至少包含 `type`、`title`、`status`、`detail`、`instance`。
- 只允许 `PATCH` 修改部分字段；不实现无意义的全量 `PUT`。

### 2.1 安装并初始化 Alembic

先阅读 Alembic 的迁移概念：ORM 模型是“当前代码期望的结构”，迁移文件是“数据库从旧结构演进到新结构的历史”。不要继续使用 `Base.metadata.create_all()` 管理生产数据表。

```python
    """表示请求的专题不存在。"""

    pass

    """表示论文已关联到该专题。"""

    pass


class TopicRepository(Protocol):

    def get(self, topic_id: UUID) -> Topic | None: ...
    def list(self, *, limit: int) -> list[Topic]: ...
    def update(self, topic_id: UUID, **values: object) -> Topic | None: ...
    def delete(self, topic_id: UUID) -> bool: ...
    def add_paper(self, link: TopicPaper) -> TopicPaper: ...
    def set_favorite(self, topic_id: UUID, paper_id: str, is_favorite: bool) -> TopicPaper | None: ...
    def list_papers(self, topic_id: UUID, *, limit: int) -> list[TopicPaperView]: ...
```

```python
from uuid import UUID

from app.domain.topics.models import Topic
from app.domain.topics.ports import TopicNotFoundError, TopicRepository


class TopicService:
    """协调专题的创建、查询与输入规范化。"""

    def __init__(self, repository: TopicRepository, workspace_id: UUID) -> None:
        self.repository = repository
        self.workspace_id = workspace_id

    @staticmethod
    def _normalize_items(values: list[str]) -> list[str]:
        result: list[str] = []
        for value in values:
            normalized = value.strip().lower()
            if normalized and normalized not in result:
                result.append(normalized)
        return result

    def create(
        self,
        name: str,
        description: str,
        keywords: list[str],
        categories: list[str],
    ) -> Topic:
        normalized_keywords = self._normalize_items(keywords)
        if not normalized_keywords:
            raise ValueError("at least one keyword is required")
        topic = Topic(
            workspace_id=self.workspace_id,
            name=name.strip(),
            description=description.strip(),
            keywords=normalized_keywords,
            categories=self._normalize_items(categories),
        )
        return self.repository.create(topic)

    def get_or_raise(self, topic_id: UUID) -> Topic:
        topic = self.repository.get(topic_id)
        if topic is None:
            raise TopicNotFoundError(str(topic_id))
        return topic

    def list(self, limit: int) -> list[Topic]:
        return self.repository.list(limit=limit)
```

```python
from uuid import UUID

from sqlalchemy import select
from sqlalchemy.exc import IntegrityError
from sqlalchemy.orm import Session

from app.domain.topics.models import Topic
from app.domain.topics.ports import TopicAlreadyExistsError
from app.infrastructure.persistence.topic_models import TopicRecord


def to_topic(record: TopicRecord) -> Topic:
    """将 ORM 记录转换为领域模型。

    :param record: 从数据库读取的专题记录。
    :return: 不依赖 SQLAlchemy 的专题领域对象。
    """

    return Topic(
        id=record.id,
        workspace_id=record.workspace_id,
        name=record.name,
        description=record.description,
        keywords=list(record.keywords),
        categories=list(record.categories),
        created_at=record.created_at,
        updated_at=record.updated_at,
    )


class PostgresTopicRepository:
    """使用 PostgreSQL 保存和读取专题。"""

    def __init__(self, session_factory) -> None:
        self.session_factory = session_factory

    def create(self, topic: Topic) -> Topic:
        record = TopicRecord(**topic.model_dump())
        try:
            with self.session_factory() as session:
                session.add(record)
                session.commit()
                session.refresh(record)
                return to_topic(record)
        except IntegrityError as exc:
            raise TopicAlreadyExistsError(topic.name) from exc

    def get(self, topic_id: UUID) -> Topic | None:
        with self.session_factory() as session:
            record = session.get(TopicRecord, topic_id)
            return None if record is None else to_topic(record)

    def list(self, *, limit: int) -> list[Topic]:
        with self.session_factory() as session:
            statement = select(TopicRecord).order_by(TopicRecord.created_at.desc()).limit(limit)
            records = session.scalars(statement).all()
            return [to_topic(record) for record in records]
```

列表按 `created_at` 倒序且只接受 `limit`。后续若确实需要游标分页，应先固定 `(created_at, id)` 排序、实现不透明 cursor 编解码，再为新增记录与相同时间戳写集成测试；在此之前不要向客户端承诺 cursor。

```python
    request: CreateTopicRequest,
    service: TopicService = Depends(get_topic_service),
) -> TopicResponse:

        topic = service.create(**request.model_dump())
    except TopicAlreadyExistsError as exc:
        raise HTTPException(status_code=409, detail="topic name already exists") from exc
    return TopicResponse.from_domain(topic)


@router.get("", response_model=TopicListResponse)
def list_topics(
    limit: int = 20,
    service: TopicService = Depends(get_topic_service),
) -> TopicListResponse:
    topics = service.list(limit=min(max(limit, 1), 100))
    return TopicListResponse(items=[TopicResponse.from_domain(topic) for topic in topics])


@router.get("/{topic_id}", response_model=TopicResponse)

    :return: 可发送给客户端的专题响应。
```

```python
from uuid import UUID

from app.domain.topics.models import TopicPaper
from app.domain.topics.ports import TopicNotFoundError, TopicPaperAlreadyExistsError


class TopicPaperService:
    """Coordinate topic-scoped paper associations and favorite state."""

    def __init__(self, topics, papers) -> None:
        self.topics = topics
        self.papers = papers

    def add(self, topic_id: UUID, paper_id: str) -> TopicPaper:
        """Link an existing global paper to one topic.

        :return: Created topic-paper association.
        """
        if self.topics.get(topic_id) is None or self.papers.get(paper_id) is None:
            raise TopicNotFoundError(str(topic_id))
        return self.topics.add_paper(TopicPaper(topic_id=topic_id, paper_id=paper_id))

    def set_favorite(self, topic_id: UUID, paper_id: str, is_favorite: bool) -> TopicPaper:
        """Update favorite state only inside the requested topic.

        :return: Updated topic-paper association.
        """
        link = self.topics.set_favorite(topic_id, paper_id, is_favorite)
        if link is None:
            raise TopicNotFoundError(f"topic paper not found: {topic_id}/{paper_id}")
        return link
```

```python
from uuid import UUID

from app.domain.topics.models import Topic
from app.domain.topics.ports import TopicNotFoundError, TopicRepository


class TopicService:
    def __init__(self, repository: TopicRepository, workspace_id: UUID) -> None:
        self.repository = repository
        self.workspace_id = workspace_id

    @staticmethod
    def _normalize_items(values: list[str]) -> list[str]:
        result: list[str] = []
        for value in values:
            normalized = value.strip().lower()
            if normalized and normalized not in result:
                result.append(normalized)
        return result

    def create(self, name: str, description: str, keywords: list[str], categories: list[str]) -> Topic:
        normalized_keywords = self._normalize_items(keywords)
        if not normalized_keywords:
            raise ValueError("at least one keyword is required")
        return self.repository.create(
            Topic(
                workspace_id=self.workspace_id,
                name=name.strip(),
                description=description.strip(),
                keywords=normalized_keywords,
                categories=self._normalize_items(categories),
            )
        )

    def get_or_raise(self, topic_id: UUID) -> Topic:
        topic = self.repository.get(topic_id)
        if topic is None:
            raise TopicNotFoundError(str(topic_id))
        return topic

    def list(self, limit: int) -> list[Topic]:
        return self.repository.list(limit=limit)

    def update(
        self,
        topic_id: UUID,
        *,
        name: str | None = None,
        description: str | None = None,
        keywords: list[str] | None = None,
        categories: list[str] | None = None,
    ) -> Topic:
        values: dict[str, object] = {}
        if name is not None:
            values["name"] = name.strip()
        if description is not None:
            values["description"] = description.strip()
        if keywords is not None:
            normalized_keywords = self._normalize_items(keywords)
            if not normalized_keywords:
                raise ValueError("at least one keyword is required")
            values["keywords"] = normalized_keywords
        if categories is not None:
            values["categories"] = self._normalize_items(categories)
        if not values:
            return self.get_or_raise(topic_id)
        topic = self.repository.update(topic_id, **values)
        if topic is None:
            raise TopicNotFoundError(str(topic_id))
        return topic

    def delete(self, topic_id: UUID) -> None:
        if not self.repository.delete(topic_id):
            raise TopicNotFoundError(str(topic_id))
```

```python
class PostgresTopicRepository:
    def __init__(self, session_factory) -> None:
        self.session_factory = session_factory

    def create(self, topic: Topic) -> Topic:
        record = TopicRecord(**topic.model_dump())
        try:
            with self.session_factory() as session:
                session.add(record)
                session.commit()
                session.refresh(record)
                return to_topic(record)
        except IntegrityError as exc:
            raise TopicAlreadyExistsError(topic.name) from exc

    def get(self, topic_id: UUID) -> Topic | None:
        with self.session_factory() as session:
            record = session.get(TopicRecord, topic_id)
            return None if record is None else to_topic(record)

    def list(self, *, limit: int) -> list[Topic]:
        with self.session_factory() as session:
            records = session.scalars(
                select(TopicRecord).order_by(TopicRecord.created_at.desc(), TopicRecord.id.desc()).limit(limit)
            ).all()
            return [to_topic(record) for record in records]

    def update(self, topic_id: UUID, **values: object) -> Topic | None:
        try:
            with self.session_factory() as session:
                record = session.get(TopicRecord, topic_id)
                if record is None:
                    return None
                for field, value in values.items():
                    setattr(record, field, value)
                session.commit()
                session.refresh(record)
                return to_topic(record)
        except IntegrityError as exc:
            raise TopicAlreadyExistsError(str(values.get("name", topic_id))) from exc

    def delete(self, topic_id: UUID) -> bool:
        with self.session_factory() as session:
            result = session.execute(delete(TopicRecord).where(TopicRecord.id == topic_id))
            session.commit()
            return result.rowcount == 1

    def add_paper(self, link: TopicPaper) -> TopicPaper:
        record = TopicPaperRecord(**link.model_dump())
        try:
            with self.session_factory() as session:
                session.add(record)
                session.commit()
                session.refresh(record)
                return to_topic_paper(record)
        except IntegrityError as exc:
            raise TopicPaperAlreadyExistsError(link.paper_id) from exc

    def get_paper_link(self, topic_id: UUID, paper_id: str) -> TopicPaper | None:
        with self.session_factory() as session:
            record = session.get(TopicPaperRecord, {"topic_id": topic_id, "paper_id": paper_id})
            return None if record is None else to_topic_paper(record)

    def set_favorite(self, topic_id: UUID, paper_id: str, is_favorite: bool) -> TopicPaper | None:
        with self.session_factory() as session:
            record = session.get(TopicPaperRecord, {"topic_id": topic_id, "paper_id": paper_id})
            if record is None:
                return None
            record.is_favorite = is_favorite
            session.commit()
            session.refresh(record)
            return to_topic_paper(record)

    def list_papers(self, topic_id: UUID, *, limit: int) -> list[TopicPaperView]:
        with self.session_factory() as session:
            statement = (
                select(PaperRecord)
                .join(TopicPaperRecord, TopicPaperRecord.paper_id == PaperRecord.paper_id)
                .where(TopicPaperRecord.topic_id == topic_id)
                .order_by(TopicPaperRecord.created_at.desc(), PaperRecord.paper_id.desc())
                .limit(limit)
            )
            return [
                TopicPaperView(
                    paper_id=record.paper_id, title=record.title, authors=list(record.authors),
                    url=record.url, is_favorite=link.is_favorite,
                )
                for record, link in session.execute(
                    select(PaperRecord, TopicPaperRecord)
                    .join(TopicPaperRecord, TopicPaperRecord.paper_id == PaperRecord.paper_id)
                    .where(TopicPaperRecord.topic_id == topic_id)
                    .order_by(TopicPaperRecord.created_at.desc(), PaperRecord.paper_id.desc()).limit(limit)
                )
            ]
```

```python
from uuid import UUID

from fastapi import APIRouter, Depends, HTTPException

from app.api.dependencies import get_rag_service, get_retrieval_service, get_topic_paper_service
from app.domain.topics.ports import TopicNotFoundError
from app.schemas.knowledge import AskResponse, SearchRequest, SearchResponse, SourceResponse

router = APIRouter(tags=["knowledge"])


def source(item: dict) -> SourceResponse:
    return SourceResponse(**{**item, "page": item.get("page") or None})


def paper_ids(topic_id: UUID, topics) -> list[str]:
    return [paper.paper_id for paper in topics.list(topic_id, limit=10_000)]


@router.post("/topics/{topic_id}/search", response_model=SearchResponse)
async def search(topic_id: UUID, request: SearchRequest, topics=Depends(get_topic_paper_service), retrieval=Depends(get_retrieval_service)) -> SearchResponse:
    try:
        sources = await retrieval.search(request.question, paper_ids=paper_ids(topic_id, topics), top_k=request.top_k)
    except TopicNotFoundError as exc:
        raise HTTPException(status_code=404, detail="topic not found") from exc
    return SearchResponse(sources=[source(item) for item in sources])


@router.post("/topics/{topic_id}/ask", response_model=AskResponse)
async def ask(topic_id: UUID, request: SearchRequest, topics=Depends(get_topic_paper_service), rag=Depends(get_rag_service)) -> AskResponse:
    try:
        answer, sources, insufficient = await rag.ask(request.question, paper_ids=paper_ids(topic_id, topics), top_k=request.top_k)
    except TopicNotFoundError as exc:
        raise HTTPException(status_code=404, detail="topic not found") from exc
    return AskResponse(answer=answer, sources=[source(item) for item in sources], insufficient_evidence=insufficient)
```

```python
from logging.config import fileConfig

from alembic import context

from app.core.config import Settings
from app.infrastructure.persistence.models import Base
from app.infrastructure.persistence import topic_models as _topic_models

config = context.config

    fileConfig(config.config_file_name)
config.set_main_option("sqlalchemy.url", Settings().postgres_url)
target_metadata = Base.metadata


def run_migrations_offline() -> None:
    context.configure(url=config.get_main_option("sqlalchemy.url"), target_metadata=target_metadata, literal_binds=True)
    with context.begin_transaction(): context.run_migrations()


def run_migrations_online() -> None:
    from sqlalchemy import engine_from_config, pool
    engine = engine_from_config(config.get_section(config.config_ini_section, {}), prefix="sqlalchemy.", poolclass=pool.NullPool)
    with engine.connect() as connection:
        context.configure(connection=connection, target_metadata=target_metadata)
        with context.begin_transaction(): context.run_migrations()


if context.is_offline_mode(): run_migrations_offline()
else: run_migrations_online()
```

```ts
async function request<T>(path: string, init?: RequestInit): Promise<T> {
  const response = await fetch(`${baseUrl}${path}`, {
    headers: { "Content-Type": "application/json", ...init?.headers },
    ...init,
  });
  if (!response.ok) {
    const body = await response.json().catch(() => null) as { detail?: string } | null;
    throw new Error(body?.detail ?? `请求失败（${response.status}）`);
  }
  return response.status === 204 ? (undefined as T) : response.json() as Promise<T>;
}
```

## 5. 常见错误与决策检查

| 现象 | 先检查什么 | 正确处理 |
| --- | --- | --- |
| 新字段未出现在数据库 | 是否只改了 ORM，未生成/执行迁移 | 生成迁移，审阅 SQL，再升级数据库。 |
| 删除专题后论文也消失 | 外键级联错误 | 只级联删除 `topic_papers`，不能删除 `papers`。 |
| Router 中出现 `Session` 或 SQL | 分层被绕过 | 将数据访问移回 Repository，Router 只调用 Service。 |
| API 路径带有 `create`、`build`、`ask` | 将动作当作资源 | 对已有资源使用标准方法；长任务建模为 Job，会话问答建模为 Message。 |
| React 页面刷新后丢失数据 | 数据仅保存在前端 state | 通过 API 保存到 PostgreSQL，再重新加载。 |

本阶段结束后，你应能清楚说明：为什么论文与专题是多对多关系、为什么需要迁移、为什么 Router 不直接操作数据库，以及 RESTful API 如何降低前后端协作成本。

```python
    id: UUID = Field(default_factory=uuid4)
    conversation_id: UUID
    role: Literal["user", "assistant"]
    content: str
    source_refs: list[dict[str, object]] = Field(default_factory=list)
```

`Literal[...]` 限制 `role` 只能是列出的字符串；`source_refs` 保存回答所用的论文、Chunk、页码和索引版本，不能只保存最终回答文本。

实现顺序是：为 `reading_status`、`conversations` 和 `messages` 创建 ORM 与迁移；用 Fake Repository 覆盖状态更新、专题隔离和消息来源保存；再注册 `PATCH /api/v1/topics/{topic_id}/papers/{paper_id}`、`POST/GET /api/v1/topics/{topic_id}/conversations`、`POST /api/v1/conversations/{conversation_id}/messages`。前端论文列表按阅读状态筛选，并在专题内恢复会话历史。只有这些 API、迁移和页面均通过测试后，才将“标记阅读状态”和“继续追问”列为阶段 1 成果。
